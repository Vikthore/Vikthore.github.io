# --- Function to calculate compressibility factor (Z) ---
def calculate_compressibility_factor(T, P, component):
  """Calculates the compressibility factor (Z) for a component using the Pitzer correlation."""

  # Get component properties from Table 5.4
  component_row = component_data[component_data['Component'] == component]
  if component_row.empty:
      raise ValueError(f"Component '{component}' not found in Table 5.4")

  w = component_row['w'].iloc[0]
  Tc = component_row['Tc'].iloc[0]
  Pc = component_row['Pc'].iloc[0]

  # Calculate reduced temperature and pressure
  Tr = (T + 273.15) / Tc  # Equation 5.41
  Pr = P / Pc  # Equation 5.42

  # Calculate B^0 and B' (Equations 5.45 and 5.46)
  B0 = 0.083 - 0.422 / (Tr**1.6)  # Equation 5.45
  B1 = 0.139 - 0.172 / (Tr**4.2)  # Equation 5.46

  # Calculate Z0 and Z' (Equations 5.43 and 5.44)
  Z0 = 1 + B0 * Pr / Tr  # Equation 5.43
  Z1 = B1 * Pr / Tr  # Equation 5.44

  # Calculate Z (Equation 5.39)
  Z = Z0 + w * Z1  # Equation 5.39

  return Z

def calculate_gas_viscosity(T, component):
    """
    Calculates the dynamic viscosity of a gas component using Sutherland's formula.

    Args:
        T: Temperature in Kelvin.
        component: The gas component (e.g., 'CH4', 'CO2', 'N2').

    Returns:
        G: Dynamic viscosity of the gas component at temperature T in Pa.s.
    """
    # Reference viscosities () at T = 293.15 K (20C) and Sutherland's constants (S)
    # for different gases (you might need to adjust these based on your specific data source)
    viscosity_data = {
        'CH4': {'mu0': 11.0e-6, 'S': 197.0},  # Methane
        'C2H6': {'mu0': 9.3e-6, 'S': 240.0},  # Ethane
        'C3H8': {'mu0': 8.0e-6, 'S': 250.0},  # Propane
        'n-C4H10': {'mu0': 7.5e-6, 'S': 300.0}, # n-Butane
        'n-C5H12': {'mu0': 7.0e-6, 'S': 340.0}, # n-Pentane
        'N2': {'mu0': 17.5e-6, 'S': 107.0},  # Nitrogen
        'CO2': {'mu0': 15.0e-6, 'S': 240.0},  # Carbon dioxide
        'H2O': {'mu0': 12.66e-6, 'S': 100} # Water (placeholder values)
        # Add more components and their data as needed
    }

    if component not in viscosity_data:
        raise ValueError(f"Component '{component}' not found in viscosity data.")

    T0 = 293.15  # Reference temperature in Kelvin (20C)
    mu0 = viscosity_data[component]['mu0']
    S = viscosity_data[component]['S']

    muG = mu0 * (T / T0)**1.5 * (T0 + S) / (T + S)  # Sutherland's formula (Equation 5.54)

    return muG

def calculate_mixture_viscosity(T, composition):
    if T <= 0:
        raise ValueError("Temperature must be positive.")
    
    sum_y_mu_sqrtM = 0
    sum_y_sqrtM = 0

    for component, mole_fraction in composition.items():
        if mole_fraction <= 0:
            continue  # Skip components with zero mole fraction
        
        mu_i = calculate_gas_viscosity(T, component)  # Calculate viscosity of individual component
        
        # Get component molecular weight from Table 5.4
        component_row = component_data[component_data['Component'] == component]
        if component_row.empty:
            raise ValueError(f"Component '{component}' not found in Table 5.4")
        M_i = component_row['Molecular_Weight'].iloc[0]

        if np.isnan(mu_i) or np.isnan(M_i):
            continue  # Skip components with undefined viscosity or molecular weight

        sum_y_mu_sqrtM += mole_fraction * mu_i * (M_i**0.5)
        sum_y_sqrtM += mole_fraction * (M_i**0.5)

    if sum_y_sqrtM <= 0:
        raise ValueError("Invalid mixture viscosity calculated.")

    mu_mix = sum_y_mu_sqrtM / sum_y_sqrtM
    return mu_mix

# --- Function to calculate fluid density ---
def calculate_fluid_density(pressure, temperature, composition):
    """Calculates fluid density using Equation 5.37b."""
    if pressure <= 0 or temperature <= -273.15:
        raise ValueError("Pressure and temperature must be positive.")

    # Convert pressure to psia and temperature to Fahrenheit
    P_psia = pressure * 14.5038  # 1 bar = 14.5038 psi
    T_F = temperature * 9/5 + 32

    rho_g = 0  # Initialize gas density
    for component, fraction in composition.items():
        if fraction <= 0:
            continue  # Skip components with zero fraction

        component_row = component_data[component_data['Component'] == component]
        if component_row.empty:
            raise ValueError(f"Component '{component}' not found in Table 5.4")
        rho_gi = component_row['Specific_Gravity'].iloc[0]

        if np.isnan(rho_gi):
            continue  # Skip components with undefined specific gravity

        # Calculate partial pressure of the component
        P_i = fraction * P_psia

        # Calculate individual Z Factor
        try:
            Zi = calculate_compressibility_factor(temperature, pressure, component)
        except ValueError as e:
            continue  # Skip this component if Z is invalid

        if Zi <= 0:
            continue  # Skip this component if Z is non-positive

        # Add the component density to the total gas density
        rho_g += (627.1047 * P_i * rho_gi) / (Zi * (460 + T_F))  # Equation 5.37b

    if rho_g <= 0:
        raise ValueError("Invalid fluid density calculated.")

    return rho_g * 16.0185  # Converting from lb/ft3 to kg/m3
